  <h1>Mapa do Cemitério Vila Formosa</h1>
  <p>Digite o número da quadra para localizar:</p>
  <input type="number" id="quadraInput" placeholder="Ex.: 56" min="1" max="100">
  <button onclick="destacarQuadra()">Localizar Quadra</button>

  <div id="mapa"></div>
  <div id="trajeto"></div>

  <script>
    const mapa = document.getElementById('mapa');
    for (let i = 1; i <= 100; i++) {
      const quadra = document.createElement('div');
      quadra.className = 'quadra';
      quadra.textContent = i;
      mapa.appendChild(quadra);
    }

    function destacarQuadra() {
      const numero = parseInt(document.getElementById('quadraInput').value);
      const quadras = document.querySelectorAll('.quadra');
      quadras.forEach(q => q.classList.remove('destacada'));

      if (numero >= 1 && numero <= 100) {
        quadras[numero - 1].classList.add('destacada');
        document.getElementById('trajeto').textContent =
          `Trajeto: Siga em linha reta até a quadra ${numero}.`;
      } else {
        document.getElementById('trajeto').textContent = 'Número de quadra inválido!';
      }
    }
  </script>
